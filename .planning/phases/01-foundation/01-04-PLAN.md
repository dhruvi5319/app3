```typescript
import { useQuery } from '@tanstack/react-query'
import type { OpenMeteoForecastResponse, ActiveLocation, WeatherData } from '../lib/types'
import { transformWeatherData } from '../lib/transforms'

// CRITICAL: timezone=auto is MANDATORY on every request — TechArch §6
const BASE_URL = 'https://api.open-meteo.com/v1/forecast'

async function fetchWeather(latitude: number, longitude: number, location: ActiveLocation): Promise<WeatherData> {
  const params = new URLSearchParams({
    latitude: String(latitude),
    longitude: String(longitude),
    timezone: 'auto',  // NON-NEGOTIABLE per TechArch §6
    current: 'temperature_2m,relative_humidity_2m,apparent_temperature,is_day,precipitation_probability,weathercode,windspeed_10m,winddirection_10m',
    hourly: 'temperature_2m,precipitation_probability,weathercode,is_day,apparent_temperature,relative_humidity_2m,uv_index,visibility',
    daily: 'weathercode,temperature_2m_max,temperature_2m_min,precipitation_probability_max,sunrise,sunset,uv_index_max,windspeed_10m_max,winddirection_10m_dominant,visibility_mean',
    forecast_days: '8',
    wind_speed_unit: 'mph',
  })

  const res = await fetch(`${BASE_URL}?${params}`)
  if (!res.ok) throw new Error(`Weather API error: ${res.status}`)

  const raw = await res.json() as OpenMeteoForecastResponse
  return transformWeatherData(raw, location)
}

export function useWeatherQuery(location: ActiveLocation | null) {
  return useQuery({
    // Cache key: ["weather", lat, lon] per TechArch §5
    queryKey: ['weather', location?.latitude, location?.longitude],
    queryFn: () => {
      if (!location) throw new Error('No active location')
      return fetchWeather(location.latitude, location.longitude, location)
    },
    // Exact config from TechArch §5
    staleTime: 10 * 60 * 1000,   // 10 minutes
    gcTime: 30 * 60 * 1000,       // 30 minutes
    retry: 2,
    retryDelay: 1_000,
    refetchOnWindowFocus: false,
    networkMode: 'offlineFirst',
    enabled: location !== null,
  })
}
```

Create `src/components/FreshnessIndicator.tsx`:

Per FRD §7: "Updated X minutes ago" always visible when data loaded. Updates every 60s.
"Updated just now" when < 1 min, "Updated 1 minute ago" for exactly 1 min, "Updated X minutes ago" for >1 min.

```tsx
import { useState, useEffect } from 'react'

interface FreshnessIndicatorProps {
  fetchedAt: number  // Unix ms timestamp from WeatherData.fetchedAt
}

export function FreshnessIndicator({ fetchedAt }: FreshnessIndicatorProps) {
  const [label, setLabel] = useState(() => computeLabel(fetchedAt))

  useEffect(() => {
    setLabel(computeLabel(fetchedAt))
    const interval = setInterval(() => {
      setLabel(computeLabel(fetchedAt))
    }, 60_000)
    return () => clearInterval(interval)
  }, [fetchedAt])

  return (
    <span
      className="text-uswds-xs text-gray-60"
      aria-live="polite"
      aria-atomic="true"
    >
      {label}
    </span>
  )
}

function computeLabel(fetchedAt: number): string {
  const minutes = Math.max(0, Math.floor((Date.now() - fetchedAt) / 60_000))
  if (minutes === 0) return 'Updated just now'
  if (minutes === 1) return 'Updated 1 minute ago'
  return `Updated ${minutes} minutes ago`
}
```

Create `src/components/CurrentConditions.tsx`:

This orchestrates loading/error/success states per FRD §3 Process and FRD §16 Never-Blank-Screen Contract.

```tsx
import type { ActiveLocation, TemperatureUnit } from '../lib/types'
import { useWeatherQuery } from '../hooks/useWeatherQuery'
import { WeatherHero } from './WeatherHero'
import { SkeletonHero } from './SkeletonHero'

interface CurrentConditionsProps {
  location: ActiveLocation
  unit: TemperatureUnit
  onUnitToggle: (unit: TemperatureUnit) => void
  onRetry: () => void
}

export function CurrentConditions({ location, unit, onUnitToggle, onRetry }: CurrentConditionsProps) {
  const { data, isLoading, isError, refetch } = useWeatherQuery(location)

  // Never-blank-screen contract: loading → skeleton, error → alert, success → data
  if (isLoading) {
    return (
      // aria-live="polite" region stays in DOM from initial render (FRD §7 validation rule)
      <div aria-live="polite" aria-atomic="true">
        <SkeletonHero />
      </div>
    )
  }

  if (isError || !data) {
    return (
      <div aria-live="assertive" role="alert">
        <div className="usa-alert usa-alert--error">
          <div className="usa-alert__body">
            <h2 className="usa-alert__heading">Weather Unavailable</h2>
            <p className="usa-alert__text">
              Unable to load weather data for {location.name}.
            </p>
            <button
              type="button"
              className="usa-button mt-3 min-h-[44px]"
              onClick={() => {
                onRetry()
                refetch()
              }}
            >
              Try again
            </button>
          </div>
        </div>
      </div>
    )
  }

  return (
    // aria-live="polite" announces weather data updates to screen readers (FRD §3 Process step 9)
    <div aria-live="polite" aria-atomic="false">
      <span className="sr-only">
        Weather updated for {location.name}: {data.current.temperatureC}°C, {/* screen reader announcement */}
      </span>
      <WeatherHero
        data={data}
        unit={unit}
        onUnitToggle={onUnitToggle}
      />
    </div>
  )
}
```
