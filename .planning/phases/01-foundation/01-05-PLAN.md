Create `src/components/AlertBanner.tsx`:

Always in DOM as `aria-live` region — empty when no alert. Per FRD §7 and §16:

```tsx
interface AlertBannerProps {
  variant: 'error' | 'warning' | 'info' | null
  heading?: string
  body?: string
  onDismiss?: () => void
  showRetry?: boolean
  onRetry?: () => void
}

export function AlertBanner({ variant, heading, body, onDismiss, showRetry, onRetry }: AlertBannerProps) {
  // Always render the aria-live wrapper — empty when no alert (FRD §7 validation rule)
  return (
    <div aria-live="polite" aria-atomic="true">
      {variant && (
        <div className={`usa-alert usa-alert--${variant}`} role="alert">
          <div className="usa-alert__body">
            {heading && <h3 className="usa-alert__heading">{heading}</h3>}
            {body && <p className="usa-alert__text">{body}</p>}
            <div className="flex gap-3 mt-2">
              {showRetry && onRetry && (
                <button
                  type="button"
                  className="usa-button min-h-[44px]"
                  onClick={onRetry}
                >
                  Retry
                </button>
              )}
              {onDismiss && (
                <button
                  type="button"
                  className="usa-button usa-button--outline min-h-[44px]"
                  onClick={onDismiss}
                >
                  Dismiss
                </button>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  )
}
```

```tsx
import { useState } from 'react'
import { AppShell } from './components/AppShell'
import { SearchBar } from './components/SearchBar'
import { CurrentConditions } from './components/CurrentConditions'
import { AlertBanner } from './components/AlertBanner'
import type { ActiveLocation, TemperatureUnit } from './lib/types'
import { getUnitPref } from './lib/storage'

export default function App() {
  const [activeLocation, setActiveLocation] = useState<ActiveLocation | null>(null)
  // Initialize unit from localStorage — no flash of wrong unit on reload (F1-04)
  const [unit, setUnit] = useState<TemperatureUnit>(() => getUnitPref())
  const [offlineAlert, setOfflineAlert] = useState<{
    variant: 'error' | 'warning' | 'info'
    heading: string
    body: string
  } | null>(null)
  const [retryKey, setRetryKey] = useState(0)

  // Offline state detection per FRD §7
  // (Full online/offline event listeners added here for completeness — 
  //  full data freshness handling is Phase 3 F7)
  // For Phase 1: basic detection only

  const handleLocationSelect = (location: ActiveLocation) => {
    setActiveLocation(location)
    setOfflineAlert(null)
  }

  const searchBar = (
    <SearchBar
      onLocationSelect={handleLocationSelect}
      hasActiveLocation={activeLocation !== null}
    />
  )

  return (
    <AppShell header={searchBar}>
      {/* Global alert banner — always in DOM (aria-live region) */}
      <AlertBanner
        variant={offlineAlert?.variant ?? null}
        heading={offlineAlert?.heading}
        body={offlineAlert?.body}
      />

      {/* Never-blank-screen: if no active location, AppShell renders centered SearchBar (via header slot) */}
      {activeLocation && (
        <div className={activeLocation ? 'pt-20' : ''}>
          {/* pt-20 offsets the fixed header when city is loaded */}
          <CurrentConditions
            key={`${activeLocation.latitude}-${activeLocation.longitude}-${retryKey}`}
            location={activeLocation}
            unit={unit}
            onUnitToggle={setUnit}
            onRetry={() => setRetryKey(k => k + 1)}
          />
        </div>
      )}
    </AppShell>
  )
}
```

**Note:** The centered SearchBar from Plan 03 fills the screen when `hasActiveLocation=false`, serving as the empty state. App.tsx doesn't need to render a separate empty state — SearchBar handles it.
