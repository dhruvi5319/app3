Create `src/hooks/useGeocoding.ts`:

This hook manages autocomplete suggestions using component-local state (NOT TanStack Query cache — FRD §12.2 says "Do not cache geocoding results in TanStack Query — they are single-use suggestions; use component-local state").

```typescript
import { useState, useEffect, useRef } from 'react'
import type { GeocodingResult } from '../lib/types'

interface UseGeocodingResult {
  suggestions: GeocodingResult[]
  isLoading: boolean
  error: string | null
  notFound: boolean
}

export function useGeocoding(query: string): UseGeocodingResult {
  const [suggestions, setSuggestions] = useState<GeocodingResult[]>([])
  const [isLoading, setIsLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [notFound, setNotFound] = useState(false)
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null)
  const abortRef = useRef<AbortController | null>(null)

  useEffect(() => {
    // Clear previous timer
    if (timerRef.current) clearTimeout(timerRef.current)
    // Clear suggestions if query too short
    if (query.trim().length < 2) {
      setSuggestions([])
      setError(null)
      setNotFound(false)
      setIsLoading(false)
      return
    }

    // 300ms debounce per FRD §2 Process step 3
    timerRef.current = setTimeout(async () => {
      // Cancel previous in-flight request
      abortRef.current?.abort()
      abortRef.current = new AbortController()
      setIsLoading(true)
      setError(null)
      setNotFound(false)

      try {
        const url = `https://geocoding-api.open-meteo.com/v1/search?name=${encodeURIComponent(query.trim())}&count=5&language=en&format=json`
        const res = await fetch(url, { signal: abortRef.current.signal })
        if (!res.ok) throw new Error(`HTTP ${res.status}`)
        const data = await res.json() as { results?: GeocodingResult[] }
        const results = data.results ?? []  // FRD §12.2: results may be absent
        setSuggestions(results.slice(0, 5))
        setNotFound(results.length === 0)
      } catch (err) {
        if (err instanceof Error && err.name === 'AbortError') return
        setError('Search unavailable — check your connection')
        setSuggestions([])
      } finally {
        setIsLoading(false)
      }
    }, 300)

    return () => {
      if (timerRef.current) clearTimeout(timerRef.current)
    }
  }, [query])

  return { suggestions, isLoading, error, notFound }
}
```

This is called imperatively (not a React hook query) — it's a one-shot function used by GpsButton. Export as a standalone async function, not a hook.

```typescript
import type { NominatimReverseResponse } from '../lib/types'

function extractCityName(response: NominatimReverseResponse, lat: number, lon: number): string {
  return response.address.city
    ?? response.address.town
    ?? response.address.village
    ?? response.address.county
    ?? `${lat.toFixed(4)}, ${lon.toFixed(4)}`
}

export async function reverseGeocode(lat: number, lon: number): Promise<string> {
  try {
    const url = `https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lon}&format=json&zoom=10&accept-language=en`
    const res = await fetch(url, {
      headers: {
        'User-Agent': 'SimpleWeatherApp/1.0',
      },
    })
    if (!res.ok) return extractCityName({ address: { country: '', country_code: '' }, place_id: 0, display_name: '' }, lat, lon)
    const data = await res.json() as NominatimReverseResponse
    return extractCityName(data, lat, lon)
  } catch {
    // Nominatim failure is silent — use coordinate fallback per FRD §2 Error States
    return `${lat.toFixed(4)}, ${lon.toFixed(4)}`
  }
}
```

Per LOCKED decision: GPS icon button sits inline at the right edge of the search input.
Per FRD §2: Denied/timeout geolocation → no error shown, search remains primary.

```tsx
import { useState } from 'react'
import type { ActiveLocation } from '../lib/types'
import { reverseGeocode } from '../hooks/useReverseGeocoding'

interface GpsButtonProps {
  onLocationDetected: (location: ActiveLocation) => void
}

export function GpsButton({ onLocationDetected }: GpsButtonProps) {
  const [isLoading, setIsLoading] = useState(false)

  const handleClick = () => {
    if (!navigator.geolocation) return
    setIsLoading(true)

    navigator.geolocation.getCurrentPosition(
      async (position) => {
        const { latitude, longitude } = position.coords
        // Validate coordinates per FRD §2
        if (!Number.isFinite(latitude) || !Number.isFinite(longitude) ||
            latitude < -90 || latitude > 90 || longitude < -180 || longitude > 180) {
          setIsLoading(false)
          return
        }
        try {
          const cityName = await reverseGeocode(latitude, longitude)
          onLocationDetected({
            id: `gps-${latitude.toFixed(4)}-${longitude.toFixed(4)}`,
            name: cityName,
            latitude,
            longitude,
          })
        } finally {
          setIsLoading(false)
        }
      },
      (_err) => {
        // Denied (code 1), unavailable (code 2), or timeout (code 3) — all silent per FRD §2
        // Geolocation timeout: show brief warning is optional but FRD says "auto-dismisses after 5s"
        // For simplicity: silent failure, search remains primary (locked decision)
        setIsLoading(false)
      },
      { timeout: 10000 }
    )
  }

  return (
    <button
      type="button"
      className="usa-button usa-button--outline min-h-[44px] min-w-[44px] px-3 flex items-center justify-center"
      aria-label="Use my location"
      onClick={handleClick}
      disabled={isLoading}
    >
      {isLoading ? (
        <span aria-hidden="true" className="text-uswds-sm">…</span>
      ) : (
        <svg aria-hidden="true" focusable="false" width="20" height="20" viewBox="0 0 24 24" fill="currentColor">
          <path d="M12 2C8.13 2 5 5.13 5 9c0 5.25 7 13 7 13s7-7.75 7-13c0-3.87-3.13-7-7-7zm0 9.5c-1.38 0-2.5-1.12-2.5-2.5s1.12-2.5 2.5-2.5 2.5 1.12 2.5 2.5-1.12 2.5-2.5 2.5z"/>
        </svg>
      )}
    </button>
  )
}
```

3. GPS icon button sits inline at right edge of search input

```tsx
import { useState, useRef, useCallback } from 'react'
import type { ActiveLocation, GeocodingResult } from '../lib/types'
import { useGeocoding } from '../hooks/useGeocoding'
import { getRecentLocations, addRecentLocation } from '../lib/storage'
import { SuggestionList } from './SuggestionList'
import { GpsButton } from './GpsButton'
import { RecentLocations } from './RecentLocations'

interface SearchBarProps {
  onLocationSelect: (location: ActiveLocation) => void
  /** True once a city has been loaded — switches from centered to fixed header */
  hasActiveLocation: boolean
}

export function SearchBar({ onLocationSelect, hasActiveLocation }: SearchBarProps) {
  const [query, setQuery] = useState('')
  const [isFocused, setIsFocused] = useState(false)
  const [activeIndex, setActiveIndex] = useState(-1)
  const inputRef = useRef<HTMLInputElement>(null)
  const listId = 'search-suggestion-list'

  const { suggestions, error: geocodingError, notFound } = useGeocoding(query)
  const recentLocations = getRecentLocations()

  const showDropdown = isFocused && (suggestions.length > 0 || notFound || (recentLocations.length > 0))
  const showSuggestions = isFocused && suggestions.length > 0

  const handleSelect = useCallback((result: GeocodingResult) => {
    const location: ActiveLocation = {
      id: String(result.id),
      name: `${result.name}${result.admin1 ? `, ${result.admin1}` : ''}, ${result.country}`,
      latitude: result.latitude,
      longitude: result.longitude,
    }
    setQuery(location.name)
    setActiveIndex(-1)
    addRecentLocation({ name: location.name, latitude: location.latitude, longitude: location.longitude, addedAt: Date.now() })
    onLocationSelect(location)
    inputRef.current?.blur()
  }, [onLocationSelect])

  const handleRecentSelect = useCallback((location: ActiveLocation) => {
    setQuery(location.name)
    addRecentLocation({ name: location.name, latitude: location.latitude, longitude: location.longitude, addedAt: Date.now() })
    onLocationSelect(location)
    inputRef.current?.blur()
  }, [onLocationSelect])

  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (!showSuggestions) return
    if (e.key === 'ArrowDown') {
      e.preventDefault()
      setActiveIndex(prev => Math.min(prev + 1, suggestions.length - 1))
    } else if (e.key === 'ArrowUp') {
      e.preventDefault()
      setActiveIndex(prev => Math.max(prev - 1, -1))
    } else if (e.key === 'Enter' && activeIndex >= 0) {
      e.preventDefault()
      const selected = suggestions[activeIndex]
      if (selected) handleSelect(selected)
    } else if (e.key === 'Escape') {
      setActiveIndex(-1)
      inputRef.current?.blur()
    }
  }

  // The search input itself — shared between centered and header states
  const searchInput = (
    <div className="relative flex items-stretch gap-0">
      <div className="relative flex-1">
        <input
          ref={inputRef}
          type="search"
          role="combobox"
          aria-expanded={showSuggestions}
          aria-controls={showSuggestions ? listId : undefined}
          aria-activedescendant={activeIndex >= 0 ? `suggestion-opt-${activeIndex}` : undefined}
          aria-autocomplete="list"
          aria-label="Search for a city"
          className="usa-input w-full min-h-[44px] text-uswds-md px-4"
          placeholder="Enter a city name"
          value={query}
          onChange={(e) => {
            setQuery(e.target.value)
            setActiveIndex(-1)
          }}
          onFocus={() => setIsFocused(true)}
          onBlur={() => {
            // Delay to allow click on suggestions to fire first
            setTimeout(() => setIsFocused(false), 200)
          }}
          onKeyDown={handleKeyDown}
          autoComplete="off"
          spellCheck={false}
        />
        {showSuggestions && (
          <SuggestionList
            suggestions={suggestions}
            activeIndex={activeIndex}
            onSelect={handleSelect}
            listId={listId}
          />
        )}
      </div>
      <GpsButton onLocationDetected={(loc) => {
        setQuery(loc.name)
        addRecentLocation({ name: loc.name, latitude: loc.latitude, longitude: loc.longitude, addedAt: Date.now() })
        onLocationSelect(loc)
      }} />
    </div>
  )

  // Inline alerts
  const alerts = (
    <>
      {notFound && (
        <div className="usa-alert usa-alert--warning mt-2" role="alert">
          <div className="usa-alert__body">
            <p className="usa-alert__text">City not found — try a different spelling</p>
          </div>
        </div>
      )}
      {geocodingError && (
        <div className="usa-alert usa-alert--error mt-2" role="alert">
          <div className="usa-alert__body">
            <p className="usa-alert__text">{geocodingError}</p>
          </div>
        </div>
      )}
    </>
  )

  // LOCKED: centered empty state for first-time users (no active location)
  if (!hasActiveLocation) {
    return (
      <div className="flex flex-col items-center justify-center min-h-screen px-4 py-16 bg-white">
        <div className="w-full max-w-lg">
          <h1 className="text-uswds-2xl font-bold text-gray-90 text-center mb-2 font-public-sans">
            Simple Weather
          </h1>
          <p className="text-uswds-md text-gray-60 text-center mb-6">
            Enter a city to get started
          </p>
          {searchInput}
          {alerts}
          <RecentLocations
            locations={recentLocations}
            onSelect={handleRecentSelect}
            visible={isFocused}
          />
          {/* aria-live region for suggestion count announcements (FRD §2) */}
          <div aria-live="polite" className="sr-only">
            {showSuggestions ? `${suggestions.length} suggestions available` : ''}
          </div>
        </div>
      </div>
    )
  }

  // LOCKED: fixed top header once city is loaded
  return (
    <header className="fixed top-0 left-0 right-0 z-40 bg-white border-b border-gray-20 shadow-sm">
      <div className="max-w-5xl mx-auto px-4 py-2">
        <div className="relative">
          {searchInput}
          {alerts}
          <RecentLocations
            locations={recentLocations}
            onSelect={handleRecentSelect}
            visible={showDropdown}
          />
          <div aria-live="polite" className="sr-only">
            {showSuggestions ? `${suggestions.length} suggestions available` : ''}
          </div>
        </div>
      </div>
    </header>
  )
}
```
