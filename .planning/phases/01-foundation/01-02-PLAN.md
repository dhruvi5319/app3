import type { TemperatureUnit, WeatherCondition, OpenMeteoForecastResponse, WeatherData, ActiveLocation, HourlySlot, DailySlot } from './types'

// §13.1 Temperature Rounding — from FRD
export function roundTemp(value: number | null | undefined): number | null {
  if (value == null || !Number.isFinite(value)) return null
  return Math.round(value)
}

export function toFahrenheit(celsius: number): number {
  return Math.round((celsius * 9) / 5 + 32)
}

export function displayTemp(celsius: number, unit: TemperatureUnit): string {
  const value = unit === 'F' ? toFahrenheit(celsius) : Math.round(celsius)
  return `${value}°${unit}`
}

// §13.2 Timezone-Aware Formatting — from TechArch §8
export function formatHourLabel(isoTime: string, timezone: string): string {
  return new Intl.DateTimeFormat('en-US', {
    hour: 'numeric',
    hour12: true,
    timeZone: timezone,
  }).format(new Date(isoTime))
}

export function formatDayLabel(dateString: string, timezone: string, index: number): string {
  if (index === 0) return 'Today'
  return new Intl.DateTimeFormat('en-US', {
    timeZone: timezone,
    weekday: 'short',
  }).format(new Date(dateString))
}

export function formatSunTime(isoTime: string, timezone: string): string {
  return new Intl.DateTimeFormat('en-US', {
    hour: 'numeric',
    minute: '2-digit',
    hour12: true,
    timeZone: timezone,
  }).format(new Date(isoTime))
}

export function getCurrentHourIndex(hourlyTimes: string[], locationTimezone: string): number {
  const now = new Date()
  const localHour = new Intl.DateTimeFormat('en-US', {
    timeZone: locationTimezone,
    hour: 'numeric',
    hour12: false,
  }).format(now)
  const idx = hourlyTimes.findIndex((t) => {
    const slotHour = new Intl.DateTimeFormat('en-US', {
      timeZone: locationTimezone,
      hour: 'numeric',
      hour12: false,
    }).format(new Date(t))
    return slotHour === localHour
  })
  return idx === -1 ? 0 : idx
}

export function getCondition(wmoCode: number, isDay: boolean): WeatherCondition & { resolvedIconId: string } {
  const base = WMO_CONDITIONS[wmoCode]
  if (!base) {
    return { iconId: 'unknown', label: 'Unknown Conditions', hasNightVariant: false, resolvedIconId: 'unknown' }
  }
  const resolvedIconId = base.hasNightVariant && !isDay
    ? base.iconId.replace('-day', '-night')
    : base.iconId
  return { ...base, resolvedIconId }
}

// Hero gradient: returns CSS custom property var() reference string
// Maps wmoCode groups to gradient CSS vars defined in index.css
export function getHeroGradientClass(wmoCode: number, isDay: boolean): string {
  if (wmoCode === 0 || wmoCode === 1) return isDay ? 'var(--hero-gradient-clear-day)' : 'var(--hero-gradient-clear-night)'
  if (wmoCode === 2) return isDay ? 'var(--hero-gradient-partly-cloudy-day)' : 'var(--hero-gradient-partly-cloudy-night)'
  if (wmoCode === 3) return isDay ? 'var(--hero-gradient-overcast-day)' : 'var(--hero-gradient-overcast-night)'
  if (wmoCode === 45 || wmoCode === 48) return 'var(--hero-gradient-fog)'
  if ((wmoCode >= 51 && wmoCode <= 67) || (wmoCode >= 80 && wmoCode <= 82)) return 'var(--hero-gradient-rain)'
  if ((wmoCode >= 71 && wmoCode <= 77) || wmoCode === 85 || wmoCode === 86) return 'var(--hero-gradient-snow)'
  if (wmoCode >= 95) return 'var(--hero-gradient-thunderstorm)'
  return isDay ? 'var(--hero-gradient-clear-day)' : 'var(--hero-gradient-clear-night)'
}

// Wind direction degrees → cardinal — from TechArch §8
const CARDINAL_DIRECTIONS = ['N','NNE','NE','ENE','E','ESE','SE','SSE',
                              'S','SSW','SW','WSW','W','WNW','NW','NNW']

export function degreesToCardinal(degrees: number): string {
  const index = Math.round(((degrees % 360) + 360) % 360 / 22.5) % 16
  return CARDINAL_DIRECTIONS[index] ?? 'N'
}

// transformWeatherData: converts raw API response to WeatherData
// Called once after TanStack Query fetch succeeds
export function transformWeatherData(
  raw: OpenMeteoForecastResponse,
  location: ActiveLocation
): WeatherData {
  const timezone = raw.timezone
  const currentHourIdx = getCurrentHourIndex(raw.hourly.time, timezone)

  const hourly: HourlySlot[] = raw.hourly.time
    .slice(currentHourIdx, currentHourIdx + 24)
    .map((t, i) => ({
      time: formatHourLabel(t, timezone),
      temperatureC: roundTemp(raw.hourly.temperature_2m[currentHourIdx + i] ?? null) ?? 0,
      precipProbability: raw.hourly.precipitation_probability[currentHourIdx + i] ?? 0,
      wmoCode: raw.hourly.weathercode[currentHourIdx + i] ?? 0,
      isDay: (raw.hourly.is_day[currentHourIdx + i] ?? 1) === 1,
    }))

  const daily: DailySlot[] = raw.daily.time.slice(0, 7).map((dateStr, i) => ({
    dayLabel: formatDayLabel(dateStr, timezone, i),
    highC: roundTemp(raw.daily.temperature_2m_max[i] ?? null) ?? 0,
    lowC: roundTemp(raw.daily.temperature_2m_min[i] ?? null) ?? 0,
    precipProbabilityMax: raw.daily.precipitation_probability_max[i] ?? 0,
    wmoCode: raw.daily.weathercode[i] ?? 0,
  }))

  const sunriseIso = raw.daily.sunrise[0] ?? ''
  const sunsetIso = raw.daily.sunset[0] ?? ''

  return {
    location,
    timezone,
    fetchedAt: Date.now(),
    current: {
      temperatureC: roundTemp(raw.current.temperature_2m) ?? 0,
      feelsLikeC: roundTemp(raw.current.apparent_temperature) ?? 0,
      humidity: raw.current.relative_humidity_2m,
      windSpeedMph: Math.round(raw.current.windspeed_10m),
      windDirectionDeg: raw.current.winddirection_10m,
      precipProbability: raw.current.precipitation_probability,
      wmoCode: raw.current.weathercode,
      isDay: raw.current.is_day === 1,
      todayHighC: roundTemp(raw.daily.temperature_2m_max[0] ?? null) ?? 0,
      todayLowC: roundTemp(raw.daily.temperature_2m_min[0] ?? null) ?? 0,
      sunrise: sunriseIso ? formatSunTime(sunriseIso, timezone) : '—',
      sunset: sunsetIso ? formatSunTime(sunsetIso, timezone) : '—',
    },
    hourly,
    daily,
    details: {
      uvIndexMax: roundTemp(raw.daily.uv_index_max?.[0] ?? null),
      windDirectionCardinal: degreesToCardinal(raw.current.winddirection_10m),
      visibilityKm: roundTemp((raw.hourly.visibility?.[currentHourIdx] ?? null) !== null
        ? (raw.hourly.visibility![currentHourIdx]! / 1000)
        : null),
      sunrise: sunriseIso ? formatSunTime(sunriseIso, timezone) : '—',
      sunset: sunsetIso ? formatSunTime(sunsetIso, timezone) : '—',
    },
  }
}

import type { RecentLocation, TemperatureUnit } from './types'

const UNIT_KEY = 'swa:unit'
const RECENT_KEY = 'swa:recent'

export function getUnitPref(): TemperatureUnit {
  const raw = localStorage.getItem(UNIT_KEY)
  return raw === 'C' || raw === 'F' ? raw : 'C'
}

export function setUnitPref(unit: TemperatureUnit): void {
  localStorage.setItem(UNIT_KEY, unit)
}

export function getRecentLocations(): RecentLocation[] {
  try {
    const raw = localStorage.getItem(RECENT_KEY)
    if (!raw) return []
    const parsed = JSON.parse(raw)
    return Array.isArray(parsed) ? parsed as RecentLocation[] : []
  } catch {
    return []
  }
}

export function addRecentLocation(location: RecentLocation): void {
  const existing = getRecentLocations().filter(
    (l) => !(Math.abs(l.latitude - location.latitude) < 0.001 && Math.abs(l.longitude - location.longitude) < 0.001)
  )
  const updated = [location, ...existing].slice(0, 5)
  localStorage.setItem(RECENT_KEY, JSON.stringify(updated))
}
